import React, { useState, useEffect, useMemo } from 'react';
import { 
  Plus, 
  Trash2, 
  CheckCircle2, 
  Circle, 
  Calendar,
  ChevronRight
} from 'lucide-react';

const App = () => {
  // State for assignments/school work
  const [assignments, setAssignments] = useState(() => {
    const saved = localStorage.getItem('school_assignments');
    return saved ? JSON.parse(saved) : [];
  });

  // State for daily tasks
  const [dailyTasks, setDailyTasks] = useState(() => {
    const saved = localStorage.getItem('daily_tasks');
    return saved ? JSON.parse(saved) : {};
  });

  // Form states
  const [assignTitle, setAssignTitle] = useState('');
  const [assignDeadline, setAssignDeadline] = useState('');
  const [newTaskText, setNewTaskText] = useState('');
  const [selectedDate, setSelectedDate] = useState(new Date().toISOString().split('T')[0]);

  // Persist data
  useEffect(() => {
    localStorage.setItem('school_assignments', JSON.stringify(assignments));
  }, [assignments]);

  useEffect(() => {
    localStorage.setItem('daily_tasks', JSON.stringify(dailyTasks));
  }, [dailyTasks]);

  // Logic to calculate the timeline range
  const timelineDates = useMemo(() => {
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    
    let endDate = new Date(today);
    endDate.setDate(endDate.getDate() + 10); 

    if (assignments.length > 0) {
      const dates = assignments.map(a => new Date(a.deadline));
      const maxDate = new Date(Math.max(...dates));
      endDate = new Date(maxDate);
      endDate.setDate(endDate.getDate() + 1);
    }

    const dateArray = [];
    let currentDate = new Date(today);
    while (currentDate <= endDate) {
      dateArray.push(new Date(currentDate).toISOString().split('T')[0]);
      currentDate.setDate(currentDate.getDate() + 1);
    }
    return dateArray;
  }, [assignments]);

  const addAssignment = (e) => {
    e.preventDefault();
    if (!assignTitle || !assignDeadline) return;
    const newAssign = {
      id: Date.now(),
      title: assignTitle,
      deadline: assignDeadline,
    };
    setAssignments([...assignments, newAssign].sort((a, b) => new Date(a.deadline) - new Date(b.deadline)));
    setAssignTitle('');
    setAssignDeadline('');
  };

  const deleteAssignment = (id) => {
    setAssignments(assignments.filter(a => a.id !== id));
  };

  const addDailyTask = (date) => {
    if (!newTaskText.trim()) return;
    const dateTasks = dailyTasks[date] || [];
    setDailyTasks({
      ...dailyTasks,
      [date]: [...dateTasks, { id: Date.now(), text: newTaskText, done: false }]
    });
    setNewTaskText('');
  };

  const toggleTask = (date, taskId) => {
    const updatedTasks = dailyTasks[date].map(t => 
      t.id === taskId ? { ...t, done: !t.done } : t
    );
    setDailyTasks({ ...dailyTasks, [date]: updatedTasks });
  };

  const removeTask = (date, taskId) => {
    const updatedTasks = dailyTasks[date].filter(t => t.id !== taskId);
    const newDailyTasks = { ...dailyTasks, [date]: updatedTasks };
    if (updatedTasks.length === 0) delete newDailyTasks[date];
    setDailyTasks(newDailyTasks);
  };

  const formatDate = (dateStr) => {
    const d = new Date(dateStr);
    return d.toLocaleDateString(undefined, { weekday: 'short', month: 'short', day: 'numeric' });
  };

  return (
    <div className="min-h-screen bg-white text-slate-800 font-sans selection:bg-slate-100 overflow-x-hidden touch-pan-y">
      {/* Meta tag simulation via script to prevent zooming on mobile */}
      <style dangerouslySetInnerHTML={{ __html: `
        /* Prevent zoom on focus for iOS */
        input, select, textarea {
          font-size: 16px !important;
        }
        body {
          touch-action: manipulation;
          overflow-x: hidden;
          width: 100%;
          position: relative;
        }
      `}} />

      <div className="max-w-3xl mx-auto px-6 pt-16 pb-24 w-full overflow-x-hidden">
        
        {/* Minimal Header */}
        <header className="mb-12">
          <h1 className="text-3xl font-light tracking-tight text-slate-900 mb-2">Workspace</h1>
          <p className="text-sm text-slate-400 font-light">Organize your academic journey and daily intentions.</p>
        </header>

        <div className="grid grid-cols-1 md:grid-cols-12 gap-12">
          
          {/* Inputs Section */}
          <div className="md:col-span-4 space-y-10">
            <section>
              <h2 className="text-xs font-semibold text-slate-400 uppercase tracking-widest mb-4">Add Work</h2>
              <form onSubmit={addAssignment} className="space-y-3">
                <input 
                  type="text" 
                  value={assignTitle}
                  onChange={(e) => setAssignTitle(e.target.value)}
                  placeholder="Task name..."
                  className="w-full bg-slate-50 border-none px-4 py-3 rounded-lg text-[16px] focus:ring-1 focus:ring-slate-200 outline-none transition-all placeholder:text-slate-300 appearance-none"
                />
                <input 
                  type="date" 
                  value={assignDeadline}
                  onChange={(e) => setAssignDeadline(e.target.value)}
                  className="w-full bg-slate-50 border-none px-4 py-3 rounded-lg text-[16px] focus:ring-1 focus:ring-slate-200 outline-none transition-all text-slate-500 appearance-none"
                />
                <button type="submit" className="w-full py-3 text-xs font-medium bg-slate-900 text-white rounded-lg hover:bg-slate-800 transition-colors active:scale-[0.98]">
                  Add Deadline
                </button>
              </form>
            </section>

            <section>
              <h2 className="text-xs font-semibold text-slate-400 uppercase tracking-widest mb-4">Deadlines</h2>
              <div className="space-y-4">
                {assignments.length === 0 && <p className="text-xs text-slate-300 italic font-light">Clear schedule.</p>}
                {assignments.map(a => (
                  <div key={a.id} className="group flex flex-col border-l border-slate-100 pl-4 py-1">
                    <div className="flex justify-between items-start">
                      <span className="text-sm font-medium text-slate-700">{a.title}</span>
                      <button onClick={() => deleteAssignment(a.id)} className="opacity-100 md:opacity-0 group-hover:opacity-100 text-slate-300 hover:text-red-400 transition-opacity p-1">
                        <Trash2 size={14} />
                      </button>
                    </div>
                    <span className="text-[10px] text-slate-400 mt-1 uppercase tracking-tighter">{formatDate(a.deadline)}</span>
                  </div>
                ))}
              </div>
            </section>
          </div>

          {/* Timeline Section */}
          <div className="md:col-span-8">
            <h2 className="text-xs font-semibold text-slate-400 uppercase tracking-widest mb-4">Timeline</h2>
            <div className="space-y-1 border-t border-slate-100">
              {timelineDates.map(date => {
                const isToday = date === new Date().toISOString().split('T')[0];
                const dateAssignments = assignments.filter(a => a.deadline === date);
                const tasks = dailyTasks[date] || [];
                const isActive = selectedDate === date;

                return (
                  <div 
                    key={date} 
                    onClick={() => setSelectedDate(date)}
                    className={`group py-5 px-4 transition-all cursor-pointer border-b border-slate-50 ${isActive ? 'bg-slate-50' : 'hover:bg-slate-50/50'}`}
                  >
                    <div className="flex items-baseline justify-between mb-2">
                      <div className="flex items-center gap-3">
                        <span className={`text-sm font-medium ${isToday ? 'text-slate-900 underline underline-offset-4 decoration-slate-300' : 'text-slate-500'}`}>
                          {formatDate(date)}
                        </span>
                        {dateAssignments.map(a => (
                          <span key={a.id} className="text-[10px] px-2 py-0.5 bg-black text-white font-bold rounded uppercase">
                            DUE: {a.title}
                          </span>
                        ))}
                      </div>
                    </div>

                    <div className="space-y-2 ml-1">
                      {tasks.map(task => (
                        <div key={task.id} className="flex items-center gap-3 group/task">
                          <button onClick={(e) => { e.stopPropagation(); toggleTask(date, task.id); }} className="p-1">
                            {task.done ? 
                              <CheckCircle2 size={16} className="text-slate-300" /> : 
                              <Circle size={16} className="text-slate-300 group-hover/task:text-slate-400" />
                            }
                          </button>
                          <span className={`text-[16px] md:text-sm ${task.done ? 'text-slate-300 line-through' : 'text-slate-600 font-light'}`}>
                            {task.text}
                          </span>
                          <button 
                            onClick={(e) => { e.stopPropagation(); removeTask(date, task.id); }}
                            className="opacity-100 md:opacity-0 group-hover/task:opacity-100 text-slate-200 hover:text-red-300 transition-opacity ml-auto p-1"
                          >
                            <Trash2 size={14} />
                          </button>
                        </div>
                      ))}

                      {isActive && (
                        <div className="mt-3 pt-2" onClick={e => e.stopPropagation()}>
                          <input 
                            autoFocus
                            type="text"
                            placeholder="Add task..."
                            value={newTaskText}
                            onChange={(e) => setNewTaskText(e.target.value)}
                            onKeyDown={(e) => e.key === 'Enter' && addDailyTask(date)}
                            className="w-full text-[16px] md:text-sm font-light bg-transparent border-none outline-none py-2 placeholder:text-slate-300 text-slate-600 appearance-none"
                          />
                        </div>
                      )}
                    </div>
                  </div>
                );
              })}
            </div>
          </div>

        </div>
      </div>
    </div>
  );
};

export default App;

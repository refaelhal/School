<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">
    <title>Workspace</title>
    <script src="https://cdn.tailwindcss.com"></script>
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <style>
        /* Prevents mobile zoom on focus and side-scrolling */
        input, select, textarea { font-size: 16px !important; }
        body { 
            touch-action: manipulation; 
            overflow-x: hidden; 
            width: 100%; 
            position: relative;
            background-color: white;
            -webkit-tap-highlight-color: transparent;
        }
        .modal-overlay {
            background-color: rgba(255, 255, 255, 0.95);
        }
        .no-scrollbar::-webkit-scrollbar { display: none; }
        
        input[type="date"]::-webkit-inner-spin-button,
        input[type="date"]::-webkit-calendar-picker-indicator {
            display: none;
            -webkit-appearance: none;
        }
    </style>
</head>
<body class="text-slate-900 font-sans selection:bg-slate-100">
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect, useMemo } = React;

        const App = () => {
            const [assignments, setAssignments] = useState(() => {
                const saved = localStorage.getItem('school_assignments');
                return saved ? JSON.parse(saved) : [];
            });

            const [dailyTasks, setDailyTasks] = useState(() => {
                const saved = localStorage.getItem('daily_tasks');
                return saved ? JSON.parse(saved) : {};
            });

            const [showAddModal, setShowAddModal] = useState(false);
            const [assignTitle, setAssignTitle] = useState('');
            const [assignDeadline, setAssignDeadline] = useState('');
            const [newTaskText, setNewTaskText] = useState('');
            const [selectedDate, setSelectedDate] = useState(new Date().toISOString().split('T')[0]);

            useEffect(() => {
                localStorage.setItem('school_assignments', JSON.stringify(assignments));
            }, [assignments]);

            useEffect(() => {
                localStorage.setItem('daily_tasks', JSON.stringify(dailyTasks));
            }, [dailyTasks]);

            const timelineDates = useMemo(() => {
                const today = new Date();
                today.setHours(0, 0, 0, 0);
                let endDate = new Date(today);
                endDate.setDate(endDate.getDate() + 14);

                if (assignments.length > 0) {
                    const dates = assignments.map(a => new Date(a.deadline));
                    const maxDate = new Date(Math.max(...dates));
                    if (maxDate > endDate) {
                        endDate = new Date(maxDate);
                        endDate.setDate(endDate.getDate() + 1);
                    }
                }

                const dateArray = [];
                let currentDate = new Date(today);
                while (currentDate <= endDate) {
                    dateArray.push(new Date(currentDate).toISOString().split('T')[0]);
                    currentDate.setDate(currentDate.getDate() + 1);
                }
                return dateArray;
            }, [assignments]);

            const addAssignment = (e) => {
                e.preventDefault();
                if (!assignTitle || !assignDeadline) return;
                const newAssign = { id: Date.now(), title: assignTitle, deadline: assignDeadline };
                setAssignments([...assignments].concat(newAssign).sort((a, b) => new Date(a.deadline) - new Date(b.deadline)));
                setAssignTitle('');
                setAssignDeadline('');
                setShowAddModal(false);
            };

            const addDailyTask = (date) => {
                if (!newTaskText.trim()) return;
                const dateTasks = dailyTasks[date] || [];
                setDailyTasks({ ...dailyTasks, [date]: [...dateTasks, { id: Date.now(), text: newTaskText, done: false }] });
                setNewTaskText('');
            };

            const formatDate = (dateStr) => {
                const d = new Date(dateStr);
                return d.toLocaleDateString(undefined, { weekday: 'short', month: 'short', day: 'numeric' });
            };

            return (
                <div className="max-w-xl mx-auto px-6 pt-16 pb-24 w-full relative">
                    
                    {/* Minimal Header */}
                    <div className="flex justify-between items-center mb-16">
                        <h1 className="text-xl font-medium tracking-tight">Study Flow</h1>
                        <button 
                            onClick={() => setShowAddModal(true)}
                            className="text-xs font-bold uppercase tracking-widest border-b border-black pb-0.5 hover:text-slate-400 hover:border-slate-400 transition-colors"
                        >
                            + New Entry
                        </button>
                    </div>

                    {/* Compact Queue */}
                    <section className="mb-16">
                        <h2 className="text-[10px] font-bold text-slate-400 uppercase tracking-[0.2em] mb-6">Upcoming</h2>
                        {assignments.length === 0 ? (
                            <p className="text-xs text-slate-300 italic">Queue clear.</p>
                        ) : (
                            <div className="space-y-4">
                                {assignments.map(a => (
                                    <div key={a.id} className="flex justify-between items-baseline group">
                                        <div className="flex items-center gap-4">
                                            <span className="text-sm font-medium">{a.title}</span>
                                            <span className="text-[10px] text-slate-400 uppercase">{formatDate(a.deadline)}</span>
                                        </div>
                                        <button onClick={() => setAssignments(assignments.filter(x => x.id !== a.id))} className="text-slate-200 hover:text-black transition-colors">
                                            <svg width="12" height="12" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="3" strokeLinecap="round" strokeLinejoin="round"><line x1="18" y1="6" x2="6" y2="18"></line><line x1="6" y1="6" x2="18" y2="18"></line></svg>
                                        </button>
                                    </div>
                                ))}
                            </div>
                        )}
                    </section>

                    {/* Streamlined Timeline */}
                    <section>
                        <h2 className="text-[10px] font-bold text-slate-400 uppercase tracking-[0.2em] mb-6">Timeline</h2>
                        <div className="space-y-px">
                            {timelineDates.map(date => {
                                const isToday = date === new Date().toISOString().split('T')[0];
                                const dateAssigns = assignments.filter(a => a.deadline === date);
                                const tasks = dailyTasks[date] || [];
                                const isActive = selectedDate === date;

                                return (
                                    <div key={date} onClick={() => setSelectedDate(date)} className={`py-6 border-b border-slate-100 transition-opacity ${!isActive && tasks.length === 0 && dateAssigns.length === 0 ? 'opacity-40' : 'opacity-100'}`}>
                                        <div className="flex items-center gap-3 mb-4">
                                            <span className={`text-xs font-bold uppercase tracking-wider ${isToday ? 'text-black' : 'text-slate-400'}`}>
                                                {formatDate(date)}
                                            </span>
                                            {dateAssigns.map(a => (
                                                <span key={a.id} className="text-[9px] font-bold text-red-500 uppercase tracking-widest">
                                                    ● {a.title}
                                                </span>
                                            ))}
                                        </div>
                                        
                                        <div className="space-y-3">
                                            {tasks.map(t => (
                                                <div key={t.id} className="flex items-center gap-4 group/task">
                                                    <div onClick={(e) => {
                                                        e.stopPropagation();
                                                        const updated = dailyTasks[date].map(x => x.id === t.id ? {...x, done: !x.done} : x);
                                                        setDailyTasks({...dailyTasks, [date]: updated});
                                                    }} className={`w-3.5 h-3.5 border transition-colors cursor-pointer ${t.done ? 'bg-black border-black' : 'border-slate-300'}`}>
                                                    </div>
                                                    <span className={`text-sm flex-1 ${t.done ? 'text-slate-300 line-through' : 'text-slate-600'}`}>{t.text}</span>
                                                    <button onClick={(e) => {
                                                        e.stopPropagation();
                                                        const filtered = dailyTasks[date].filter(x => x.id !== t.id);
                                                        setDailyTasks({...dailyTasks, [date]: filtered});
                                                    }} className="text-slate-100 hover:text-black opacity-0 group-hover/task:opacity-100 transition-opacity">
                                                        <svg width="12" height="12" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="3" strokeLinecap="round" strokeLinejoin="round"><line x1="18" y1="6" x2="6" y2="18"></line><line x1="6" y1="6" x2="18" y2="18"></line></svg>
                                                    </button>
                                                </div>
                                            ))}
                                            {isActive && (
                                                <div className="pl-7" onClick={e => e.stopPropagation()}>
                                                    <input 
                                                        placeholder="Add task..."
                                                        value={newTaskText}
                                                        onChange={(e) => setNewTaskText(e.target.value)}
                                                        onKeyDown={(e) => e.key === 'Enter' && addDailyTask(date)}
                                                        className="w-full bg-transparent border-none outline-none text-sm py-1 placeholder:text-slate-200 text-slate-500 appearance-none"
                                                    />
                                                </div>
                                            )}
                                        </div>
                                    </div>
                                );
                            })}
                        </div>
                    </section>

                    {/* Minimal Modal */}
                    {showAddModal && (
                        <div className="fixed inset-0 z-[100] modal-overlay flex items-center justify-center p-6" onClick={() => setShowAddModal(false)}>
                            <div className="bg-white w-full max-w-xs p-8 border border-slate-100 shadow-sm" onClick={e => e.stopPropagation()}>
                                <div className="flex justify-between items-center mb-8">
                                    <h2 className="text-xs font-bold uppercase tracking-widest">Entry</h2>
                                    <button onClick={() => setShowAddModal(false)} className="text-slate-300 hover:text-black transition-colors">✕</button>
                                </div>
                                <form onSubmit={addAssignment} className="space-y-8">
                                    <div className="space-y-1">
                                        <input 
                                            type="text" 
                                            value={assignTitle}
                                            onChange={(e) => setAssignTitle(e.target.value)}
                                            placeholder="Subject Name"
                                            className="w-full border-b border-slate-100 py-2 text-sm outline-none focus:border-black transition-colors placeholder:text-slate-200"
                                        />
                                    </div>
                                    <div className="space-y-1">
                                        <input 
                                            type="date" 
                                            value={assignDeadline}
                                            onChange={(e) => setAssignDeadline(e.target.value)}
                                            className="w-full border-b border-slate-100 py-2 text-sm outline-none focus:border-black transition-colors text-slate-500"
                                        />
                                    </div>
                                    <button type="submit" className="w-full py-3 bg-black text-white text-[10px] font-bold uppercase tracking-[0.2em] hover:bg-slate-800 transition-colors">
                                        Confirm
                                    </button>
                                </form>
                            </div>
                        </div>
                    )}
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
</html>
